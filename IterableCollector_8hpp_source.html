144  void collect () override
145  {
146  if (SPARTA_EXPECT_FALSE(isCollected()))
147  {
148  if(SPARTA_EXPECT_FALSE(iterable_object_.size() > expected_capacity_))
149  {
150  if(SPARTA_EXPECT_FALSE(warn_on_size_))
151  {
152  sparta::log::MessageSource::getGlobalWarn()
153  << "WARNING! The collected object '"
154  << getLocation() << "' has grown beyond the "
155  << "expected capacity (given at construction) for collection. "
156  << "Expected " << expected_capacity_ << " but grew to "
157  << iterable_object_.size()
158  << " This is your first and last warning.";
159  warn_on_size_ = false;
160  }
161  }
162  collectImpl_(std::integral_constant<bool, sparse_array_type>());
163  }
164  }

194  // Standard walk of iterable types
195  void collectImpl_(std::false_type)
196  {
197  auto itr = iterable_object_.begin();
198  auto eitr = iterable_object_.end();
199  for (uint32_t i = 0; i < expected_capacity_; ++i)
200  {
201  if (itr != eitr) {
202  positions_[i]->collect(*itr);
203  ++itr;
204  } else {
205  positions_[i]->closeRecord();
206  }
207  }
208  }

210  // Full iteration walk, checking validity of the iterator. This
211  // is used for Pipe and Array where the iterator points to valid
212  // and not valid entries in the component
213  void collectImpl_(std::true_type)
214  {
215 
216  uint32_t s = 0;
217  auto itr = iterable_object_.begin();
218  for (uint32_t i = 0; i < expected_capacity_; ++i, ++s)
219  {
220  if (itr.isValid()) {
221  positions_[i]->collect(*itr);
222  } else {
223  positions_[i]->closeRecord();
224  }
225  ++itr;
226  }
227  }
