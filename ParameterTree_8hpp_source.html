390  Node* create(const std::string& path, bool required) {
391  if(path.size() == 0){
392  return this;
393  }
394 
395  size_t name_pos = 0;
396  std::string immediate_child_name;
397  immediate_child_name = TreeNode::getNextName(path, name_pos);
398  if(immediate_child_name.size() == 0){
399  // Cannot get parent like this
400  return nullptr; // TEMPORARY behavior. See docstring
401  //throw SpartaException("Virtual parameter path \"") << path
402  // << "\" is invalid because it contains an empty name (between two '.' "
403  // "characters). Parents cannot currently be referenced in the virtual parameter tree";
404  }
405 
406  // Get a child of node if one exists with an exact match before any nodes
407  // with wilcards are encountered. If this node name includes a wildcard, a
408  // child will be returned IFF it the pattern string itself matches with the
409  // highest priority child in this node.
410  Node* child = getPriorityChildMatch(immediate_child_name);
411  if(!child){
412  child = addChild(immediate_child_name, required);
413  }
414 
415  if(name_pos == std::string::npos){
416  return child;
417  }
418 
419  const std::string remainder = path.substr(name_pos);
420  return child->create(remainder, required);

431  Node* getPriorityChildMatch(const std::string& name) const {
432  if(children_.size() == 0){
433  return nullptr;
434  }
435 
436  // Becase the name is a pattern, it is impossible to math against other existing
437  // children which are patterns. Therefore, if a pattern child is encountered
438  // which does not exactly match the given name string, nullptr must be returned so
439  // that the caller appends a new node, whih will have the highest priority
440  const bool name_has_wildcard = TreeNode::hasWildcardCharacters(name);
441 
442  // Always search in reverse-applied order to match on most recent changes first.
443  //
444  // This is an optimization to prevent nodes from being created needlessly. It tries
445  // to find a matching node while considering wildcards to avoid creating new nodes
446  // in most cases where it is possible.
447  //
448  // Read the rules below to understand what happens. In general, if we don't get an
449  // exact string match with an existing node before hitting another node which is
450  // an exclusive superset or incomplete subset of the 'name' argument's pattern (it
451  // may have no wildcards) then a new node will need to be added at the end of the
452  // children list to specify a value. An existing node cannot be overridden.
453  auto itr = children_.rbegin();
454  for(; itr != children_.rend(); ++itr){
455  if((*itr)->getName() == name){
456  // Found a node with an exact name match (no pattern matching) before
457  // hitting a pattern node. This node can be used to apply a new parameter
458  return itr->get();
459  }
460 
461  if(name_has_wildcard){
462  if(TreeNode::hasWildcardCharacters((*itr)->getName())){
463  // Encountered a wildcard node whch was not an exact string match.
464  // No way to tell if the name pattern exactly matches this node's, so
465  // assume it doesn't
466  return nullptr;
467  }else if(matches(name, (*itr)->getName())){
468  // Node has not wildcards but matches the pattern in 'name'. Therfore,
469  // A new parameter with this name would affect this node and more. A new
470  // node will have to be added to override it.
471  return nullptr;
472  }else{
473  // Node has no wildcards and the name agument is a pattern which
474  // does not match it. Therefore, it can be ignored because this 'name'
475  // will not apply
476  continue;
477  }
478  }else if(matches((*itr)->getName(), name)){
479  // Encountered a wildcard node which matches on this name before hitting an
480  // exact match. Therefore, a new node must be created by the caller so that
481  // the parameter being set will affect a subset of this node's pattern
482  return nullptr;
483  }
484  }
485  // No matches or important mismatches
486  return nullptr;
487  }

616  // this node if it has the same patch or a matching pattern.

623  if (n) {
624  n = n->getParent();
625  if (n && n->getName() == "extension") {
626  return false;

1100  template<typename NodeT>
1101  static NodeT* recursTryGet_(NodeT* node, const std::string& path,
1102  const std::string& match_name,
1103  size_t name_pos, const bool must_be_leaf)
1104  {
1105  sparta_assert(!TreeNode::hasWildcardCharacters(match_name),
1106  "Cannot attempt to read a node with a path containing wildcard "
1107  "characters. A specific node path must be used. Error in \""
1108  << match_name << "\" from \"" << path << "\"");
1109 
1110  if(name_pos == std::string::npos){
1111  // End of the search.. No deeper
1112  NodeT* result = nullptr;
1113  NodeT* backup = nullptr; // First match (if is has no value)
1114  auto itr = node->getMatcherBegin();
1115  for(; itr != node->getMatcherEnd(); ++itr){
1116  if(itr.matches(match_name)){
1117  if(itr.get()->hasValue() || !must_be_leaf){
1118  itr.get()->incrementReadCount();
1119  if(!result){
1120  result = itr.get(); // Found it
1121  }
1122  }else if(!backup){
1123  backup = itr.get();
1124  }
