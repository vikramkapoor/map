729  void invalidateIndexes_() {
730  // To invalidate any and all outstanding iterators, move
731  // the start/end indexes outside the current window. Do
732  // not set the start_idx_ to the old end_idx_ as any older
733  // "end" iterator would be considered valid (equals the
734  // start_idx_)
735  start_idx_ = ++end_idx_;
736  for(auto & it : circularbuffer_data_) {
737  it.window_idx = end_idx_;
738  ++end_idx_;
739  }
740  }
