164  void verifyRecords(const std::string & db_file) const {
165  simdb::ObjectManager obj_mgr(".");
166  if (!obj_mgr.connectToExistingDatabase(db_file)) {
167  return;
168  }
169 
170  auto numeric_db = GET_DB_FROM_CURRENT_SIMULATION(NumericMeta);
171  if (numeric_db) {
172  auto values_query =
173  numeric_db->createObjectQueryForTable("Numbers");
174 
175  if (values_query) {
176  double first = 0, second = 0;
177  values_query->writeResultIterationsTo(
178  "First", &first, "Second", &second);
179 
180  if (values_query->countMatches() != indices_schemaA_.size()) {
181  throw sparta::SpartaException("Could not verify SimDB records");
182  }
183 
184  auto result_iter = values_query->executeQuery();
185  size_t record_idx = 0;
186  while (result_iter->getNext()) {
187  const auto & expected = records_schemaA_[indices_schemaA_[record_idx]];
188  if (first != expected.numbers.First) {
189  throw sparta::SpartaException("Could not verify SimDB records");
190  }
191  if (second != expected.numbers.Second) {
192  throw sparta::SpartaException("Could not verify SimDB records");
193  }
194  ++record_idx;
195  }
196  }
197 
198  auto meta_query =
199  numeric_db->createObjectQueryForTable("Metadata");
200  if (meta_query) {
201  std::string name;
202  double value = 0;
203  meta_query->writeResultIterationsTo("Name", &name, "Value", &value);
204 
205  if (meta_query->countMatches() != indices_schemaA_.size()) {
206  throw sparta::SpartaException("Could not verify SimDB records");
207  }
208 
209  auto result_iter = meta_query->executeQuery();
210  size_t record_idx = 0;
211  while (result_iter->getNext()) {
212  const auto & expected = records_schemaA_[indices_schemaA_[record_idx]];
213  if (name != expected.metadata.Name) {
214  throw sparta::SpartaException("Could not verify SimDB records");
215  }
216  if (value != expected.metadata.Value) {
217  throw sparta::SpartaException("Could not verify SimDB records");
218  }
219  ++record_idx;
220  }
221  }
222  }
223  }
224 
225  private:
226  std::vector<TestSQLiteSchemaA> records_schemaA_;
227  std::vector<TestSQLiteSchemaB> records_schemaB_;
228  std::vector<TestHDF5SchemaC> records_schemaC_;
229  std::vector<uint16_t> indices_schemaA_;
230  std::vector<uint16_t> indices_schemaB_;
231  std::vector<uint16_t> indices_schemaC_;

237  // Schema builder to test two simdb::ObjectManager's
238  // bound to the same database file, separated in that
239  // same file by their respective application name.
240  // A third schema builder is for another ObjectManager,
241  // though it will be used to write records to an HDF5
242  // database, and therefore will be in its own file.
243  // SimDB's worker thread should be able to keep them
244  // separated into two groups: one group for the two
245  // SQLite database connections, and one group only
246  // serving the one HDF5 connection.
247  //
248  // Note that the two schema builders below have some
249  // overlap in their table definitions: schemaA and
250  // schemaB have some of the same table names, but
251  // these tables have different column configurations.
252  // This should not be a problem for ObjectManager
253  // since it will use its unique application name
254  // with the table names we give it to create a
255  // unique schema inside the shared file, separated
256  // from other applications tied to the same file.
257  // The specific way in which the schemas are kept
258  // separate in the file is not our concern; the
259  // DbConnProxy subclasses take care of those
260  // specifics.
261  void buildSchemaA(simdb::Schema & schema)
262  {
263  using dt = simdb::ColumnDataType;
264 
265  schema.addTable("Numbers")
266  .addColumn("First", dt::double_t)
267  .addColumn("Second", dt::double_t);
268 
269  schema.addTable("Metadata")
270  .addColumn("Name", dt::string_t)
271  .addColumn("Value", dt::double_t);
272  }
273 
274  void buildSchemaB(simdb::Schema & schema)
275  {
276  using dt = simdb::ColumnDataType;
277 
278  schema.addTable("Strings")

281 
282  schema.addTable("Metadata")

366 template <typename DataT>
367 void validateParameter(const sparta::ParameterSet & params,
368  const std::string & param_name,
369  const DataT & expected_value)
370 {
371  if (!params.hasParameter(param_name)) {
372  return;
373  }
374  const DataT actual_value = params.getParameterValueAs<DataT>(param_name);
375  if (actual_value != expected_value) {
376  throw sparta::SpartaException("Invalid extension parameter encountered:\n")
377  << "\tParameter name: " << param_name
378  << "\nParameter value (actual): " << actual_value
379  << "\nParameter value (expected): " << expected_value;
380  }
381 }

383 template <typename DataT>
384 void validateParameter(const sparta::ParameterSet & params,
385  const std::string & param_name,
386  const std::set<DataT> & expected_values)
387 {
388  bool found = false;
389  for (const auto & expected : expected_values) {
390  try {
391  found = false;
392  validateParameter<DataT>(params, param_name, expected);
393  found = true;
394  break;
395  } catch (...) {
396  }
397  }
398 
399  if (!found) {
400  throw sparta::SpartaException("Invalid extension parameter "
401  "encountered for '") << param_name << "'";
402  }

546 void ExampleSimulator::buildTree_()
547 {
548  // TREE_BUILDING Phase. See sparta::PhasedObject::TreePhase
549  // Register all the custom stat calculation functions with (cycle)histogram nodes
550  registerStatCalculationFcns_();
551 
552  auto cpu_factory = getCPUFactory_();
553 
554  // Set the cpu topology that will be built
555  cpu_factory->setTopology(cpu_topology_, num_cores_);
556 
557  // Create a single CPU
558  sparta::ResourceTreeNode* cpu_tn = new sparta::ResourceTreeNode(getRoot(),
559  "cpu",
560  sparta::TreeNode::GROUP_NAME_NONE,
561  sparta::TreeNode::GROUP_IDX_NONE,
562  "CPU Node",
563  cpu_factory);
564  to_delete_.emplace_back(cpu_tn);
565 
566  // Tell the factory to build the resources now
567  cpu_factory->buildTree(getRoot());
568 
569  // Print the registered factories
570  if(show_factories_){
571  std::cout << "Registered factories: \n";
572  for(const auto& f : getCPUFactory_()->getResourceNames()){
573  std::cout << "\t" << f << std::endl;
574  }
575  }
576 
577  // Validate tree node extensions during tree building
578  for(uint32_t i = 0; i < num_cores_; ++i){
579  sparta::TreeNode * dispatch = getRoot()->getChild("cpu.core0.dispatch", false);
580  if (dispatch) {
581  sparta::TreeNode::ExtensionsBase * extensions = dispatch->getExtension("user_data");
582 
583  // If present, validate the parameter values as given in the extension / configuration file
584  if (extensions != nullptr) {
585  const sparta::ParameterSet * dispatch_prms = extensions->getParameters();
586  sparta_assert(dispatch_prms != nullptr);
587  validateParameter<std::string>(*dispatch_prms, "when_", "buildTree_");
588  validateParameter<std::string>(*dispatch_prms, "why_", "checkAvailability");
589  }
590 
591  // There might be an extension given in --extension-file that is not found
592  // at all in any --config-file given at the command prompt. Verify that if
593  // present, the value is as expected.
594  extensions = dispatch->getExtension("square");
595  if (extensions != nullptr) {
596  const sparta::ParameterSet * dispatch_prms = extensions->getParameters();
597  sparta_assert(dispatch_prms != nullptr);
598  validateParameter<std::string>(*dispatch_prms, "edges_", "4");
599  }
600  }
601 
602  // See if there are any extensions for the alu0/alu1 nodes
603  sparta::TreeNode * alu0 = getRoot()->getChild("cpu.core0.alu0");
604  sparta::TreeNode * alu1 = getRoot()->getChild("cpu.core0.alu1");
605  if (alu0) {
606  sparta::TreeNode::ExtensionsBase * extensions = alu0->getExtension("difficulty");
607  if (extensions != nullptr) {
608  const sparta::ParameterSet * alu0_prms = extensions->getParameters();
609  sparta_assert(alu0_prms != nullptr);
610 
611  validateParameter<std::string>(*alu0_prms, "color_", "black");
612  validateParameter<std::string>(*alu0_prms, "shape_", "diamond");
613  }
614  }
615  if (alu1) {
616  sparta::TreeNode::ExtensionsBase * extensions = alu1->getExtension("difficulty");
617  if (extensions != nullptr) {
618  const sparta::ParameterSet * alu1_prms = extensions->getParameters();
619  sparta_assert(alu1_prms != nullptr);
620 
621  validateParameter<std::string>(*alu1_prms, "color_", "green");
622  validateParameter<std::string>(*alu1_prms, "shape_", "circle");
623  }
624  }
625 
626  // Once again, ask for a named extension for a tree node that was just created.
627  // The difference here is that the 'circle' extension also has a factory associated
628  // with it.
629  sparta::TreeNode * fpu = getRoot()->getChild("cpu.core0.fpu", false);
630  if (fpu) {
631  sparta::TreeNode::ExtensionsBase * extensions = fpu->getExtension("circle");
632 
633  // If present, validate the parameter values as given in the extension / configuration file
634  if (extensions != nullptr) {
635  const sparta::ParameterSet * fpu_prms = extensions->getParameters();
636  sparta_assert(fpu_prms != nullptr);
637 
638  validateParameter<std::string>(*fpu_prms, "color_", "green");
639  validateParameter<std::string>(*fpu_prms, "shape_", "round");
640  validateParameter<double> (*fpu_prms, "degrees_", 360.0);
641

1099  return;
