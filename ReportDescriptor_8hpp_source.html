655  void push_back(const ReportDescriptor & rd) {
656  //Internal book-keeping is only for Python shell workflows.
657  //We also do not perform any book-keeping for descriptors
658  //whose dest_file is "1" (print to stdout). That is the
659  //only use case where duplicate dest_file's is actually
660  //okay. We will add the descriptor to the collection, but
661  //any attempts to access this descriptor later (i.e. to
662  //change its parameters or to disable it from Python) will
663  //throw an exception that the descriptor "1" cannot be found.
664  if (rd.dest_file != "1") {
665  const std::string desc_name = getDescriptorName_(rd);
666  auto iter = indices_by_descriptor_name_.find(desc_name);
667  if (iter != indices_by_descriptor_name_.end()) {
668  //Descriptor found by this name. Throw if it is already
669  //enabled. Users can only replace disabled descriptors.
670  if (rep_descs_.at(iter->second).isEnabled()) {
671  throw SpartaException("Report descriptor named '") <<
672  desc_name << "' already exists in this configuration";
673  }
674  }
675  indices_by_descriptor_name_[desc_name] = rep_descs_.size();
676  }
677 
678  //This incoming 'rd' variable could be a reference to a
679  //previously disabled descriptor:
680  //
681  // >>> rd = report_config.descriptors.foo_csv
682  // >>> report_config.removeReport('foo_csv')
683  // >>> report_config.addReport(rd)
684  //
685  //This would still count as an enabled descriptor.
686  rep_descs_.push_back(rd);
687  rep_descs_.back().enabled_ = true;
688  }

722  bool contains(const std::string & desc_name) const {
723  auto iter = indices_by_descriptor_name_.find(desc_name);
724  if (iter == indices_by_descriptor_name_.end()) {
725  //The name may have been given to us with a period, not
726  //an underscore, separating the file stem and the extension

797  size_t getDescriptorIndexByName_(const std::string & desc_name) const {
798  auto iter = indices_by_descriptor_name_.find(desc_name);
799  if (iter == indices_by_descriptor_name_.end()) {
800  //The name may have been given to us with a period, not
801  //an underscore, separating the file stem and the extension
802  const auto replaced_desc_name = replaceDotsWithUnderscores_(desc_name);
803  iter = indices_by_descriptor_name_.find(replaced_desc_name);
804 
805  //If it still is not found, error out
806  if (iter == indices_by_descriptor_name_.end()) {
807  throw SpartaException(
808  "No descriptor named '") << desc_name << "' exists";
809  }
810  }
811 
812  //The descriptor exists in this collection, but we still need
813  //to throw if it has been disabled, since we are not supposed
814  //to be using those descriptors for any reason anymore.
815  if (!rep_descs_.at(iter->second).isEnabled()) {
816  throw SpartaException("The descriptor named '") << desc_name
817  << "' has already been disabled";
818  }
819  return iter->second;
820  }
