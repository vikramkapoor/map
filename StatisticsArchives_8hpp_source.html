32  explicit StatisticsArchives(const std::string & db_dir)
33  {
34  namespace bfs = boost::filesystem;
35  bfs::path p(db_dir);
36 
37  if (!bfs::is_directory(p)) {
38  throw SpartaException(
39  "The path given is not a directory: ") << db_dir;
40  }
41 
42  //Archives are organized as follows:
43  //
44  // db_directory
45  // db_subdirectory <-> foo.csv
46  // db_subdirectory <-> bar.json
47  // ... ...
48  //
49  //We need to loop over the subdirectories, and create
50  //a RootArchiveNode object for each one.
51  std::vector<bfs::directory_entry> subdirs;
52  std::copy(bfs::directory_iterator(p),
53  bfs::directory_iterator(),
54  std::back_inserter(subdirs));
55 
56  auto dir_iter = subdirs.begin();
57  while (dir_iter != subdirs.end()) {
58  const std::string archive_fulldir = dir_iter->path().string();
59  createArchivePlaceholderForExistingDatabase_(archive_fulldir);
60  ++dir_iter;
61  }
62  }

126  void loadArchiveFspartaxistingDatabase_(const std::string & archive_fulldir)
127  {
128  std::shared_ptr<RootArchiveNode> root(new RootArchiveNode);
129 
130  const std::string meta_filename = archive_fulldir + "/archive_tree.bin";
131  std::ifstream fin(meta_filename, std::ios::binary);
132  if (!fin) {
133  throw SpartaException(
134  "Unable to open archive file for read: ") << meta_filename;
135  }
136 
137  boost::archive::binary_iarchive ia(fin);
138  ia >> *root;
139 
140  //Give everyone in this archive tree easy access to their
141  //raw values filename
142  const std::string binary_filename = archive_fulldir + "/values.bin";
143  root->setMetadataValue("output_filename", binary_filename);
144 
145  //Give the root archive node a controller it can use to
146  //save the archive to another directory. Offline controllers
147  //do not implement synchronization.
148  std::shared_ptr<ArchiveController> controller(
149  new OfflineArchiveController(archive_fulldir));
150  root->setArchiveController(controller);
151 
152  //Make the connection from child nodes to their parent node
153  recursSetParentForChildNodes_(root.get());
154 
155  //The archive directories given to us are in the form:
156  // "db_directory/db_subdirectory"
157  //
158  //The unique archive name is simply the db_subdirectory, which
159  //will be something like 'foo.csv', 'bar.html', etc.
160  std::vector<std::string> split;
161  boost::split(split, archive_fulldir, boost::is_any_of("/"));
162  const std::string archive_name = split.back();
163  addHierarchyRoot(archive_name, root);
164  }
