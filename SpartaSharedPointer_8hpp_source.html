626  template<typename ...PointerTArgs>
627  MemBlock * allocate_(PointerTArgs&&... args)
628  {
629  // Return memory allocated here.
630  MemBlock * block = nullptr;
631 
632  // Check for previously freed blocks and reuse them.
633  if(free_idx_ > 0) {
634  --free_idx_;
635  block = free_blocks_[free_idx_];
636  sparta_assert(block->ref_count->p != nullptr);
637  block->ref_count->count = 1;
638  // perform an in place new on the reused pointer
639  new (block->ref_count->p) PointerT(std::forward<PointerTArgs>(args)...);
640  }
641  else {
642  if(SPARTA_EXPECT_FALSE(allocated_ > water_mark_)) {
643  if(SPARTA_EXPECT_FALSE(!water_mark_warning_)) {
644  watermark_warning_callback_(*this);
645  water_mark_warning_ = true;
646  }
647 
648  // Only need to check for overallocation if we've passed the watermark
649  if(SPARTA_EXPECT_FALSE(allocated_ >= memory_blocks_.capacity())) {
650  SpartaException ex;
651  ex << "This allocator has run out of memory: \n\n\t"
652  << __PRETTY_FUNCTION__
653  << "\n\n"
654  << "\t\tNumber blocks preallocated: " << memory_blocks_.capacity()
655  << "\n\t\tWatermark : " << water_mark_;
656  throw ex;
657  }
658  }
659  block = memory_blocks_.allocate(this, std::forward<PointerTArgs>(args)...);
660  ++allocated_;
661  }
662  return block;
663  }
