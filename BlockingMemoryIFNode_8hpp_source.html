406  virtual bool tryWrite(addr_t addr,
407  addr_t size,
408  const uint8_t *buf,
409  const void *in_supplement=nullptr,
410  void *out_supplement=nullptr) override {
411  ++*ctr_writes_;
412  if(__builtin_expect(doesAccessSpan(addr, size), 0)){
413  // BlockingMemoryIF base class claims that tryRead and tryWrite cannot throw
414  // throw MemoryWriteError(addr, size, "addr is in a different block than addr+size");
415  return false;
416  }
417  if(__builtin_expect(isInAccessWindows(addr, size) == false, 0)){
418  return false;
419  }
420  if(__builtin_expect(post_write_noti_.observed() == false, 1)){
421  return tryWrite_(addr, size, buf, in_supplement, out_supplement); // Write and return if unobserved
422  }
423 
424  // peek into block-size buffer
425  if(!tryPeek(addr, size, prior_val_buffer_.get())){
426  return false;
427  }
428 
429  bool result = tryWrite_(addr, size, buf, in_supplement, out_supplement);
430 
431  if(result){
432  PostWriteNotiSrc::data_type post_write_noti_data(this);
433  post_write_noti_data.addr = addr;
434  post_write_noti_data.size = size;
435  post_write_noti_data.prior = prior_val_buffer_.get();
436  post_write_noti_data.tried = buf;
437  post_write_noti_data.in_supplement = in_supplement;
438  // A getInternalBuffer method seems appropriate here but is
439  // not used because no assumptions should be made about
440  // storage. So Memory implementation may not HAVE an
441  // internal buffer for this memory (e.g. memory mapped registers not
442  // contiguous)
443  // Instead, caller must peek themselves.
444  post_write_noti_.postNotification(post_write_noti_data);
