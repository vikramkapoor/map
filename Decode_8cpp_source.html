78  // Decode instructions
79  void Decode::decodeInsts_()
80  {
81  uint32_t num_decode = std::min(uop_queue_credits_, fetch_queue_.size());
82  num_decode = std::min(num_decode, num_to_decode_);
83 
84  if(num_decode > 0)
85  {
86  InstGroup insts;
87  // Send instructions on their way to rename
88  for(uint32_t i = 0; i < num_decode; ++i) {
89  insts.emplace_back(fetch_queue_.read(0));
90 
91  if(SPARTA_EXPECT_FALSE(info_logger_)) {
92  info_logger_ << "Decoded inst: " << fetch_queue_.read(0);
93  }
94 
95  fetch_queue_.pop();
96  }
97 
98  // Send decoded instructions to rename
99  uop_queue_outp_.send(insts);
100 
101  // Decrement internal Uop Queue credits
102  uop_queue_credits_ -= num_decode;
103 
104  // Send credits back to Fetch to get more instructions
105  fetch_queue_credits_outp_.send(num_decode);
106  }
107 
108  // If we still have credits to send instructions as well as
109  // instructions in the queue, schedule another decode session
110  if(uop_queue_credits_ > 0 && fetch_queue_.size() > 0) {
111  ev_decode_insts_event_.schedule(1);
112  }
113  }
