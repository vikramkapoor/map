181  void runScheduler_()
182  {
183  // Start simulation -- align the schedulers
184  sparta_assert(sparta_scheduler_->nextEventTick() > 0);
185 
186  // Align the schedulers. Sparta starts at tick 1
187  wait(sc_core::sc_time(double(1), sparta_sc_time_));
188 
189  do {
190  // If the Sparta Scheduler has nothing to do, put it to sleep
191  if(sparta_scheduler_->nextEventTick() == Scheduler::INDEFINITE) {
192  sparta_scheduler_->registerForNotification<Scheduler::Tick,
193  SysCSpartaSchedulerAdapter,
194  &SysCSpartaSchedulerAdapter::wakeupAdapter_>(this, "item_scheduled");
195  wait(sc_wake_sparta_);
196  }
197 
198  //
199  // Wait for SysC to get to the next event time on the
200  // Sparta Scheduler, then advance Sparta
201  //
202  const sc_core::sc_time sysc_time = sc_core::sc_time_stamp();
203  if(sparta_scheduler_->nextEventTick() >= sysc_time.value()) {
204  // Wait until the sysc scheduled catches up with Sparta
205  wait(sc_core::sc_time(double(sparta_scheduler_->nextEventTick() - sysc_time.value()),
206  sparta_sc_time_));
207  }
208 
209  // Align to the posedge events in systemc
210  wait(sc_core::SC_ZERO_TIME);
211 
212  // If given, schedule the user's SystemC query event to
213  // allow the Sparta user to check to see if the SystemC side
214  // of simulation is complete. This query can be as simple
215  // as asking the SystemC kernel if there are any events
216  // (sc_core::sc_pending_activity()) or asking the SystemC
217  // simulator if it's done.
218  if(sysc_query_event_ && sparta_scheduler_->getCurrentTick() >= next_sysc_event_fire_tick_)
219  {
220  next_sysc_event_fire_tick_ = sparta_scheduler_->getCurrentTick() + sysc_query_event_interval_;
221  sysc_query_event_->scheduleRelativeTick(1, sparta_scheduler_);
222  }
223 
224  advanceSpartaScheduler_();
225 
226  } while(!sparta_scheduler_->isFinished() || !sysc_simulation_done_);
227 
228  // Stop simulation
229  sc_core::sc_stop();
230  sc_stop_called_ = true;
231  }

233  void advanceSpartaScheduler_()
234  {
235  const sc_core::sc_time sysc_time = sc_core::sc_time_stamp();
236 
237  // The SystemC scheduler will always be exactly at the same
238  // tick as Sparta when this function is called. Following
239  // this rule allows safe assumptions in scheduling
240  // synchronization.
241  sparta_assert(sysc_time.value() == sparta_scheduler_->nextEventTick());
242  sparta_assert(sparta_scheduler_->nextEventTick() >= sparta_scheduler_->getCurrentTick());
243 
244  const bool exacting_run = true;
245  const bool measure_scheduler_time = false; // no need to do this
246  // Run to the next event scheduled and then run that cycle
247  sparta_scheduler_->run(sparta_scheduler_->nextEventTick() - sparta_scheduler_->getCurrentTick() + 1,
248  exacting_run, measure_scheduler_time);
249 
250  // Sparta should now be finished with the cycle at `sysc_time`
251  // and be exactly one cycle ahead of it.
252  sparta_assert(sysc_time.value() + 1 == sparta_scheduler_->getCurrentTick());
253  }
