246  {

284  Clock::Cycle computeNextAvailableCycleForSend(sparta::Clock::Cycle send_delay_cycles, const uint32_t num_beats)
285  {
286  sparta_assert(sync_in_port_ != 0, "Attempting to send data on unbound port:" << getLocation());
287  sparta_assert(clk_->isPosedge(), "Posedge check failed in port:" << getLocation());
288 
289  // Start at the current clock with the specified delay
290  Clock::Cycle current_cycle = clk_->currentCycle();
291  Scheduler::Tick current_tick = clk_->currentTick();
292 
293  // Send each beat of data with a slide, so know previous slide
294  Scheduler::Tick prev_data_arrival_tick =
295  sync_in_port_->getPrevDataArrivalTick_();
296 
297  // Send one extra beat (<=) to represent the N+1 beat to be sent
298  for(uint32_t beat_count = 0; beat_count <= num_beats; ++beat_count) {
299  uint64_t sched_delay_ticks =
300  sync_in_port_->computeSendToReceiveTickDelay_(clk_,
301  send_delay_cycles + beat_count,
302  true /*allow_slide*/,
303  prev_data_arrival_tick);
304  prev_data_arrival_tick = current_tick + sched_delay_ticks;
305  }
306 
307  // At this point, the prev_data_arrival_tick is the absolute
308  // tick at which an N+1 beat would arrive. Now, find when that
309  // would have been sent to have that arrival time.
310  uint64_t num_ticks_before_arrival =
311  sync_in_port_->computeReverseSendToReceiveTickDelay_(clk_,
312  send_delay_cycles,
313  prev_data_arrival_tick);
314  uint64_t send_tick = prev_data_arrival_tick - num_ticks_before_arrival;
315 
316  // Convert the absolute tick of the send event into a current
317  // cycle relative cycle and return that value
318  Clock::Cycle next_send_cycle = clk_->getCycle(send_tick);
319  sparta_assert(next_send_cycle > current_cycle);
320  Clock::Cycle delay_cycle = next_send_cycle - current_cycle;
321 
322  return delay_cycle;
323  }

656  {
657  sparta::Scheduler::Tick cur_tick =
658  scheduler_->getCurrentTick();
659 
660  sparta_assert(set_ready_tick_ <= cur_tick, "Assert in port: " << getLocation());
661 
662  // If the owner of the Inport isn't ready, then re-send the
663  // data to ourself until the owner is ready to accept.
664  sparta_assert(num_in_flight_ > 0);
665  num_in_flight_--;

671  }
672  bool allow_slide = false;
673  bool is_fwd_progress = false;
674 
675  send_(dat, receiver_clock_, sparta::Clock::Cycle(0), allow_slide, is_fwd_progress);
676  sparta_assert(num_in_flight_ > 0);
677  }
678

680  else {
681  DataContainer<DataT>::setData_(dat);
682

686  explicit_consumer_handler_((const void*)&dat);

688 
689  // Show the data that has arrived on this OutPort that
690  // the receiver now sees
691  if(SPARTA_EXPECT_FALSE(collector_ != nullptr)) {
692  collector_->collectWithDuration(dat, 1);
693  }

752 
753  sparta::Scheduler::Tick cur_tick =
754  scheduler_->getCurrentTick();
755 
756  sparta::Scheduler::Tick abs_scheduled_tick =
757  num_delay_ticks + cur_tick;
758 
759  bool retval = (abs_scheduled_tick > prev_data_arrival_tick_ || prev_data_arrival_tick_ == PREV_DATA_ARRIVAL_TICK_INIT);
760

764  // 'ready' if downstream indicated it couldn't take data
765  if (getLatchedReady_(cur_tick) == false) {
766  sparta_assert(send_clk->getFrequencyMhz() == receiver_clock_->getFrequencyMhz(), "Error in port:" << getLocation());

769  "Ready/Valid only tested for zero and one cycle delays (not "<< getPortDelay() <<"); relax this assert once more testing is done; location=" << getLocation());

773  // (since we need to deliver on the same cycle its sent)
774  if (num_in_flight_ > 0 || getPortDelay() == 0) {
775  retval = false;
776  }

801  uint64_t computeSendToReceiveTickDelay_(const Clock *send_clk,
802  const double send_delay_cycles,
803  const bool allow_slide,
804  const Scheduler::Tick prev_data_arrival_tick) const
805  {
806  Scheduler::Tick num_delay_ticks =
807  calculateClockCrossingDelay(send_clk->getTick(send_delay_cycles), send_clk, receive_delay_ticks_, receiver_clock_);
808 
809  sparta::Scheduler::Tick current_tick = scheduler_->getCurrentTick();
810  sparta::Scheduler::Tick abs_scheduled_tick = num_delay_ticks + current_tick;
811 
812  // Slide pushes this send out past the previous arrival,
813  // rather than faulting as user error on sending too early
814  if (allow_slide &&
815  ( abs_scheduled_tick <= prev_data_arrival_tick &&
816  prev_data_arrival_tick != PREV_DATA_ARRIVAL_TICK_INIT)) {
817  abs_scheduled_tick = prev_data_arrival_tick + receiver_clock_->getPeriod();
818  }
819 
820  num_delay_ticks = abs_scheduled_tick - current_tick;
821 
822  // Underlying assumption is that all destinations get their
823  // event at the same time.
824  sparta_assert((abs_scheduled_tick % receiver_clock_->getPeriod()) == 0, "Failed posedge check in:" << getLocation()); // posedge check
825 
826  return num_delay_ticks;
827  }
