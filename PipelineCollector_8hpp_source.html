221  // Initialize the clock/collectable map and find the fastest clock
222  const sparta::Clock* fastest_clk = nullptr;
223  std::function<void (const sparta::Clock*)> addClks;
224  addClks = [&addClks, this, &fastest_clk] (const sparta::Clock* clk)
225  {
226  if(clk != nullptr){
227  auto & u_p = clock_ctn_map_[clk];
228  for(uint32_t i = 0; i < NUM_SCHEDULING_PHASES; ++i) {
229  u_p[i].reset(new CollectablesByClock(clk, static_cast<SchedulingPhase>(i)));
230  }
231  for(const sparta::TreeNode* child : sparta::TreeNodePrivateAttorney::getAllChildren(clk)) {
232  const sparta::Clock* child_clk = dynamic_cast<const sparta::Clock*>(child);
233  if(child_clk){
234  auto clk_period = child_clk->getPeriod();
235  // If this clock has a non-1 period (i.e. not the root clock)
236  // AND there is either
237  // (A) no fastest clock yet
238  // or
239  // (B) this clock is a higher frequency than the fastest clock.
240  // then choose this as the fastest
241  if(clk_period != 1 && (!fastest_clk || (clk_period < fastest_clk->getPeriod()))) {
242  fastest_clk = child_clk;
243  }
244  addClks(child_clk);
245  }
246  }
247  }
248  };
249  addClks(root_clk);
250  if(fastest_clk == nullptr){
251  fastest_clk = root_clk;
252  }
253 
254  // A multiple to multiply against the fastest clock when no heartbeat was set.
257  static const uint32_t heartbeat_multiplier = 200;
258  // round heartbeat using a multiple of the fastest clock if one was not set.
259  if (heartbeat_interval == 0)
260  {
261  sparta_assert(fastest_clk->getPeriod() != 0);
262  heartbeat_interval = fastest_clk->getPeriod() * heartbeat_multiplier; //round up to the nearest multiple of 100
263  heartbeat_interval = heartbeat_interval + (100 - (heartbeat_interval % 100));
264  // pipeViewer requires that intervals be a multiple of 100.
265  sparta_assert(heartbeat_interval % 100 == 0)
266  }
267 
268  // We are gonna subtract one from the heartbeat_interval
269  // later.. Better be greater than one.
270  sparta_assert(heartbeat_interval > 1);
271  // Initialize some values.
272  filepath_ = filepath;
273  heartbeat_interval_ = heartbeat_interval;
274  closing_time_ = heartbeat_interval;
275  root_clk_ = root_clk;
276 
277  ev_heartbeat_.setContinuing(false); // This event does not keep simulation going
278  }

322  void startCollection(sparta::TreeNode* starting_node)
323  {
324  if(collection_active_ == false)
325  {
326  // Create the outputter used for writing transactions to disk.
327  writer_.reset(new pipeViewer::Outputter(filepath_, heartbeat_interval_));
328 
329  // We need to write an index on the start BEFORE any transactions have been written.
330  writer_->writeIndex();
331 
332  // Write the clock information out
333  writeClockFile_();
334 
335  // Open the locations file
336  location_writer_.reset(new pipeViewer::LocationFileWriter(filepath_));
337 
338  // The reader needs heartbeat indexes up to the current
339  // collection point. This can happen on delayed pipeline
340  // collection. Start the heartbeats at the first interval
341  // as the Outputter class handles hb 0
342  last_heartbeat_ = 0;
343  const uint64_t num_hb = scheduler_->getCurrentTick()/heartbeat_interval_;
344  uint64_t cnt = 0;
345  while(cnt != num_hb) {
346  // write an index
347  writer_->writeIndex();
348  last_heartbeat_ += heartbeat_interval_;
349  ++cnt;
350  }
351 
352  // Schedule a heartbeat at the next interval, offset from
353  // the current tick. For example, if the scheduler is at
354  // 6,600,456, then we want to schedule at 7,000,000 if the
355  // interval is 1M and the num_hb is 6
356  ev_heartbeat_.scheduleRelativeTick(((num_hb + 1) * heartbeat_interval_) -
357  scheduler_->getCurrentTick(), scheduler_);
358 
359  collection_active_ = true;
360  }
361 
362  *(location_writer_.get()) << (*starting_node);

364  // Recursively collect the start node and children
365  std::function<void (sparta::TreeNode* starting_node)> recursiveCollect;
366  recursiveCollect = [&recursiveCollect, this] (sparta::TreeNode* starting_node)
367  {

369  CollectableTreeNode* c_node = dynamic_cast<CollectableTreeNode*>(starting_node);
