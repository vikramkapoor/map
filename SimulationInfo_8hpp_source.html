326  SimulationInfo(const simdb::ObjectManager & sim_db,
327  const simdb::DatabaseID obj_mgr_db_id,
328  const simdb::DatabaseID report_node_id = 0) :
329  SimulationInfo()
330  {
331  sim_db.safeTransaction([&]() {
332  simdb::ObjectQuery global_meta_query(sim_db, "SimInfo");
333 
334  global_meta_query.addConstraints(
335  "ObjMgrID", simdb::constraints::equal, obj_mgr_db_id);
336 
337  global_meta_query.writeResultIterationsTo(
338  "Name", &sim_name_,
339  "Cmdline", &command_line_,
340  "Exe", &executable_,
341  "SimulatorVersion", &simulator_version_,
342  "Repro", &reproduction_info_,
343  "SpartaVersion", &sparta_version_,
344  "Start", &start_time_);
345 
346  auto result_iter = global_meta_query.executeQuery();
347  if (!result_iter->getNext()) {
348  throw SpartaException("Unable to locate a SimInfo record with ")
349  << "ObjMgrID equal to " << obj_mgr_db_id;
350  }

352  //Clear out any strings that were "unset" in the database.
353  //SimDB's "unset" is equal to std::string's empty().
354  auto clear_unset_string = [](std::string & str) {
355  if (str == "unset") {
356  str.clear();
357  }
358  };
359  clear_unset_string(sim_name_);
360  clear_unset_string(command_line_);
361  clear_unset_string(executable_);
362  clear_unset_string(simulator_version_);
363  clear_unset_string(reproduction_info_);
364  clear_unset_string(start_time_);
365 
366  //Apply any report-specific metadata we find.
367  if (report_node_id > 0) {
368  simdb::ObjectQuery report_meta_query(sim_db, "RootReportNodeMetadata");
369 
370  report_meta_query.addConstraints(
371  "ReportNodeID", simdb::constraints::equal, report_node_id,
372  "Name", simdb::constraints::equal, "Elapsed");
373 
374  std::string elapsed;

376  result_iter = report_meta_query.executeQuery();
377  if (result_iter->getNext()) {
