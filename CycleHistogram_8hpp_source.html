91  double getStandardDeviation() const {
92 
93  // Summation of counts in normal bins plus underflow bin and
94  // overflow bin.
95  const double sum = std::accumulate(bin_.begin(), bin_.end(), 0.0)
96  + (*underflow_bin_) + (*overflow_bin_);
97 
98  // Total number of bins is number of regular bins
99  // plus one for underflow bin and plus one for
100  // overflow bin.
101  const std::size_t total_num_bins = bin_.size() + 2;
102 
103  const double mean = sum / total_num_bins;
104  double accum = 0.0;
105  std::for_each(bin_.begin(), bin_.end(), [&](const sparta::CycleCounter& c) {
106  auto c_get = c.get();
107  accum += pow(c_get - mean, 2);
108  });
109  accum += pow((*underflow_bin_ - mean), 2);
110  accum += pow((*overflow_bin_ - mean), 2);
111  return std::sqrt(accum / (total_num_bins - 1));
112  }
