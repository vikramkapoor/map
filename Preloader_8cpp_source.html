18  void Preloader::preloadPacket_(const std::string& treenode,
19  sparta::cache::PreloadPkt& pkt)
20  {
21  // This is a very dump preloader and just sends the pkt directly to
22  // the Preloadable node specified in the yaml. Other preloaders
23  // may be more verbose.
24  sparta::RootTreeNode* root = getContainer()->getSimulation()->getRoot();
25  std::vector<sparta::TreeNode*> nodes;
26  root->getSearchScope()->findChildren(treenode, nodes);
27  bool preloaded_atleast_one = false;
28  for (auto& node : nodes)
29  {
30  sparta::cache::PreloadableIF* cache = dynamic_cast<sparta::cache::PreloadableIF*>(node);
31  if (cache)
32  {
33  bool success = cache->preloadPkt(pkt);
34  preloaded_atleast_one |= success;
35  }
36  }
37  sparta_assert(preloaded_atleast_one, "Failed to preload the packet to any cache");
38  }
