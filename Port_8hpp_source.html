566  void bind(Port * in) override
567  {
568  if(in->getDirection() != Port::IN) {
569  throw SpartaException("ERROR: Attempt to bind an outty: '" + in->getName() +
570  "' to an outty: '" + getName() + "'");
571  }
572 
573  if(!sync_port_) {
574  sparta_assert(in->getClock()->getFrequencyMhz() == getClock()->getFrequencyMhz(),
575  "Trying to bind two ports that are on different clocks with different freq. "
576  "Recommend using SyncPorts");
577  }
578 
579  sparta_assert(!isAlreadyBound(in),
580  "Port: '" << getLocation()
581  << "' is already bound to '" << in->getLocation() << "' ");
582 
583  InPort * inp = nullptr;
584  if((inp = dynamic_cast<InPort *>(in)) == 0) {
585  throw SpartaException("ERROR: Could not case '" +
586  in->getName() + "' to an InPort for some reason...");
587  }
588 
589  // This outport precedes the newly binded in port if the
590  // delay of the inport is zero and the modeler intends to
591  // use the OutPort as zero delay.
592  //
593  // The precedence that's established is the following, but
594  // only if the producer, the inport's internal events, and
595  // the consumer event are all in the same phase.
596  //
597  // ,-----------------------------------------------.
598  // | V
599  // producer -> [ inport internal delivery ]* -> consumer
600  // [ inport handler delivery (optional) ]
601  //
602  //
603  // * The inport -> consumer precedence is established
604  // during the consumer registration from a call to
605  // InPort::registerConsumerEvent()
606  //
607  if((inp->getPortDelay() == 0) && presume_zero_delay_) {
608  for(auto & pd : port_producers_)
609  {
610  for(auto & cons : inp->port_consumers_) {
611  sparta_assert(pd != cons,
612  "Somehow, someway, '" << pd->getLabel() << "' is registered "
613  << "as a producer of Port: '" << getLocation()
614  << "' and, at the same time, a consumer of Port: '"
615  << inp->getLocation() << "'");
616  if(pd->getSchedulingPhase() == cons->getSchedulingPhase()) {
617  std::string reason = "Port::bind(" + getLocation() + "->" + inp->getLocation() + ")";
618  pd->precedes(cons, reason);
619  }
620  }
621  inp->setProducerPrecedence_(pd);
622  }
623  }
624  bound_ports_.push_back(inp);
625  inp->setContinuing(continuing_);
626  inp->bind_(this);
627  }
