611  void erase(const uint32_t& idx)
612  {
613  // Make sure we are invalidating an already valid object.
614  sparta_assert(idx < size(), "Cannot erase an index that is not already valid");
615 
616  // Do the invalidation immediately
617  // 1. Move the free space pointer to the erased position.
618  // 2. Call the DataT's destructor
619  // 3. Set the current free space pointer's next to the old free position
620  DataPointer* oldFree = free_position_;
621  free_position_ = buffer_map_[idx];
622  free_position_->data->~value_type();
623  free_position_->next_free = oldFree;
624 
625  // Mark DataPointer as invalid
626  validator_->detachDataPointer(free_position_);
627 
628  // Shift all the positions above the invalidation in the map one space down.
629  uint32_t i = idx;
630  sparta_assert(num_valid_ > 0);
631  const uint32_t top_idx_of_buffer = num_valid_ - 1;
632  while(i < top_idx_of_buffer)
633  {
634  // assert that we are not going to do an invalid read.
635  sparta_assert(i + 1 < num_entries_);
636  buffer_map_[i] = buffer_map_[i + 1];
637  buffer_map_[i]->physical_idx = i;
638 
639  // Shift the indexes in the address map.
640  address_map_[i] = address_map_[i + 1];
641  ++i;
642  }
643 
644  // the entry at the old num_valid_ in the map now points to nullptr
645  buffer_map_[top_idx_of_buffer] = nullptr;
646 
647  // Remove this entry of the address map as it becomes a free position.
648  address_map_.erase(top_idx_of_buffer);
649 
650  // update counts.
651  --num_valid_;
652  updateUtilizationCounters_();
653  }

863  void resizeInternalContainers_() {
864 
865  // Assert that the Buffer class is in Infinite-Mode.
866  sparta_assert(is_infinite_mode_, "The Buffer class must be in Infinite-Mode in order to resize itself.");
867 
868  // We do not resize if there are available slots in buffer.
869  if(numFree() != 0) {
870  return;
871  }
872 
873  // Resize the buffer_map_ with the amount provided by user.
874  buffer_map_.resize(buffer_map_.capacity() + resize_delta_);
875 
876  // The number of entries the buffer can hold is its capacity.
877  num_entries_ = buffer_map_.capacity();
878 
879  // Resize the data_pool_ to twice the capacity of the buffer_map_.
880  data_pool_.resize(num_entries_ * 2);
881 
882  // The number of entries the pool can hold is its capacity.
883  data_pool_size_ = data_pool_.capacity();
884 
885 
886  // Each entry in data_pool_ should have their next free position
887  // pointer point to the slot to its right.
888  for(uint32_t i = 0; i < data_pool_size_ - 1; ++i) {
889  data_pool_[i].next_free = &data_pool_[i + 1];
890  }
891 
892  // The last entry should point to itself as the next free position.
893  data_pool_[data_pool_size_ - 1].next_free = &data_pool_[data_pool_size_ - 1];
894 
895  // The first position should point to the first entry.
896  first_position_ = &data_pool_[0];
897 
898  // The free position should point to the location according to
899  // the number of entries in the buffer.
900  free_position_ = &data_pool_[num_valid_];
901 
902  // Make all the pointers in buffer_map_ point to the appropriate indexes.
903  for(uint32_t i = 0; i < num_valid_; ++i) {
904  buffer_map_[i] = &data_pool_[address_map_[i]];
905  }
906 
907  // Resize the validator vector and relink the validator data pool.
908  validator_->resizeIteratorValidator(resize_delta_, data_pool_);
909  }
