259  void registerHandlerAtStage(const uint32_t & id, const SpartaHandler & handler)
260  {
261  sparta_assert(static_cast<uint32_t>(default_precedence_) == static_cast<uint32_t>(Precedence::NONE),
262  "You have specified a default precedence (" << static_cast<uint32_t>(default_precedence_)
263  << ") between stages. No new handlers can be registered any more!");
264  sparta_assert(id < event_list_at_stage_.size(),
265  "Attempt to register handler for invalid pipeline stage[" << id << "]!");
266 
267  // Create a new stage event handler, and add it to its event list
268  auto & event_list = event_list_at_stage_[id];
269  event_list.emplace_back(new sparta::UniqueEvent<sched_phase>(&es_,
270  "uev_stage_" + std::to_string(id) + "_" + std::to_string(event_list_at_stage_[id].size()),
271  handler));
272  auto new_event = event_list.back().get();
273 
274  // Set clock for this new event handler
275  new_event->setScheduleableClock(clock_);
276  new_event->setScheduler(clock_->getScheduler());
277 
278  // Update event matrix
279  auto & event_list_per_phase = event_matrix_at_stage_[id][static_cast<uint32_t>(sched_phase)];
280  if (event_list_per_phase.size() > 0) {
281  auto & producer_event = event_list_per_phase.back();
282 
283  // Set the latest registered event (in this 'sched_phase') to precedes this new event
284  producer_event->precedes(*new_event);
285  } else {
286  events_valid_at_stage_[id] = true;
287  }
288 
289  // Add the raw pointer of newly added event to the stage event list on 'sched_phase'
290  event_list_per_phase.push_back( new_event );

301  void setPrecedenceBetweenStage(const uint32_t & pid, const uint32_t & cid)
302  {
303  sparta_assert(static_cast<uint32_t>(default_precedence_) == static_cast<uint32_t>(Precedence::NONE),
304  "You have specified a default precedence (" << static_cast<uint32_t>(default_precedence_)
305  << "). No more precedence between stages can be set!");
306  sparta_assert(pid != cid, "Cannot specify precedence with yourself!");
307  sparta_assert((pid < event_list_at_stage_.size()) && (event_list_at_stage_[pid].size() > 0),
308  "Precedence setup fails: No handler for pipeline stage[" << pid << "]!");
309  sparta_assert((cid < event_list_at_stage_.size()) && (event_list_at_stage_[cid].size() > 0),
310  "Precedence setup fails: No handler for pipeline stage[" << cid << "]!");
311 
312  for (uint32_t phase_id = 0; phase_id < NUM_SCHEDULING_PHASES; phase_id++) {
313  auto & pstage_event_list = event_matrix_at_stage_[pid][phase_id];
314  auto & cstage_event_list = event_matrix_at_stage_[cid][phase_id];
315 
316  if (pstage_event_list.empty() || cstage_event_list.empty()) {
317  continue;
318  }
319 
320  // The last event in producer pipeline stage(#pid) is
321  // scheduled earlier than the first event in consumer
322  // pipeline stage(#cid)
323  (pstage_event_list.back())->precedes(*(cstage_event_list.front()));

364  void setDefaultStagePrecedence(const Precedence & default_precedence)
365  {
366  sparta_assert(static_cast<uint32_t>(default_precedence) < static_cast<uint32_t>(Precedence::NUM_OF_PRECEDENCE),
367  "Unknown default precedence is specified for sparta::Pipeline!");
368 
369  if (static_cast<uint32_t>(default_precedence) == static_cast<uint32_t>(Precedence::NONE)) {
370  return;
371  }
372 
373  bool forward = (static_cast<uint32_t>(default_precedence) == static_cast<uint32_t>(Precedence::FORWARD));
374 
375  uint32_t pid = 0;
376  uint32_t cid = pid + 1;
377  while (cid < event_list_at_stage_.size()) {
378  if (event_list_at_stage_[pid].empty()) {
379  ++pid;
380  } else if (event_list_at_stage_[cid].empty()) {
381  ++cid;
382  } else {
383  if (forward) {
384  // This function can only be called when 'default_precedence_' is Precedence::NONE
385  setPrecedenceBetweenStage(pid, cid);
386  } else {
387  setPrecedenceBetweenStage(cid, pid);
388  }
389 
390  // Skip the empty stages between producer and consumer to avoid duplicated checking
391  pid = cid;
392  }
393 
394  if (pid == cid) {
395  ++cid;
396  }
397  }
398 
399  default_precedence_ = default_precedence;
400  }

806  void drain_(const uint32_t start_id)
807  {
808  uint32_t stage_id = num_stages_ - 1;
809 
810  // remove last item from pipe
811  if (start_id <= stage_id && pipe_.isValid(stage_id)) {
812  pipe_.invalidatePS(stage_id);
813  }
814 
815  // advance stages after stall point
816  while (stage_id > start_id) {
817  if (pipe_.isValid(stage_id - 1)) {
818  pipe_.writePS(stage_id, pipe_.access(stage_id - 1));
819  pipe_.invalidatePS(stage_id - 1);
820  }
821 
822  --stage_id;
823  }
824 
825  // check stages up to stall point for bubbles
826  // NOTE: stage_id now points to 1 past the stall point
827  // The stall point cannot advance because its events are disabled
828  // The stage before the stall cannot advance because the stall stage is occupied
829  // So, there are two useless loops here
830  while (stage_id > 0) {
831  if (advance_into_stage_[stage_id - 1] &&
832  !pipe_.isValid(stage_id) && pipe_.isValid(stage_id - 1)) {
833  pipe_.writePS(stage_id, pipe_.access(stage_id - 1));
834  pipe_.invalidatePS(stage_id - 1);
835  }
836 
837  --stage_id;
838  }
839 
840  // try advancement from insertion point
841  pipe_.shiftAppend();
842  }
