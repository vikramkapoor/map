429 /*
430  * Return the delay, in ticks, incurred when crossing a clock boundary
431  *
432  * \param src_delay The sender's delay (in ticks) to schedule from "now"
433  * \param src_clk The sender's clock
434  * \param dst_delay The receiver's delay (in ticks) to schedule from "now"
435  * \param dst_clk The receiver's clock
436  * \note both clocks must be on the same scheduler. See
437  * sparta::Clock::getScheduler
438  *
439  * \return Relative scheduler tick that the crossing would incur
440  */
441 inline Scheduler::Tick calculateClockCrossingDelay(Scheduler::Tick src_delay, const Clock * src_clk,
442  Scheduler::Tick dst_delay, const Clock * dst_clk)
443 {
444  sparta_assert(src_clk, "calculateClockCrossingDelay requires a non-null src_clk");
445  sparta_assert(dst_clk, "calculateClockCrossingDelay requires a non-null dst_clk");
446  sparta_assert(src_clk->getScheduler() == dst_clk->getScheduler(),
447  "calculateClockCrossingDelay requires src_clk and dst_clk to operate on "
448  "the same scheduler. src = " << src_clk->getScheduler() << " and dst = "
449  << dst_clk->getScheduler());
450  auto scheduler = src_clk->getScheduler();
451  sparta_assert(scheduler,
452  "calculateClockCrossingDelay requires src_clk (" << *src_clk << ") to "
453  "have a non-null scheduler");
454 
455  sparta::Scheduler::Tick current_tick = scheduler->getCurrentTick();
456  sparta::Scheduler::Tick num_delay_ticks = 0;
457 
458  // 1. Snap to source positive edge (assert match)
459  // 2. Add source delay in source clock
460  // 3. Add destination delay in source clock
461  // 4. Snap to destination positive edge
462 
463  // Note: A snap to destination positive edge between source delay and
464  // destination delay is not required. If dst_delay is integer, it has
465  // no effect. If dst_delay is float, it acts as implicit round up.
466  // Similarly, a snap to source positive edge would have the same effect
467  // of being benign or rounding up the src_delay.
468 
469  sparta::Scheduler::Tick last_src_clk_posedge_tick =
470  current_tick / src_clk->getPeriod() * src_clk->getPeriod();
471  sparta_assert(last_src_clk_posedge_tick == current_tick);
472 
473  num_delay_ticks += src_delay + dst_delay;
474 
475  sparta::Scheduler::Tick raw_event_arrival_tick = (current_tick + num_delay_ticks);
476  sparta::Scheduler::Tick raw_dst_clk_posedge_tick =
477  raw_event_arrival_tick / dst_clk->getPeriod() * dst_clk->getPeriod();
478  if (raw_event_arrival_tick != raw_dst_clk_posedge_tick) {
479  sparta_assert(raw_event_arrival_tick > raw_dst_clk_posedge_tick);
480  num_delay_ticks += dst_clk->getPeriod() - (raw_event_arrival_tick - raw_dst_clk_posedge_tick);
481  }
482 
483  return num_delay_ticks;
484 }

486 /*
487  * Return the delay, in ticks, incurred when crossing a clock boundary in
488  * the reverse direction
489  *
490  * \param dst_arrival_tick The tick in the future on which the receiver receives the data
491  * \param src_delay The sender's delay (in ticks) to schedule from "now"
492  * \param src_clk The sender's clock
493  * \param dst_delay The receiver's delay (in ticks) to schedule from "now"
494  * \param dst_clk The receiver's clock
495  * \note both clocks must be on the same scheduler. See
496  * sparta::Clock::getScheduler
497  *
498  * \return Relative scheduler tick that the reverse crossing would incur
499  */
500 inline Scheduler::Tick calculateReverseClockCrossingDelay(Scheduler::Tick dst_arrival_tick,
501  Scheduler::Tick src_delay, const Clock * src_clk,
502  Scheduler::Tick dst_delay, const Clock * dst_clk)
503 {
504  sparta_assert(src_clk, "calculateReverseClockCrossingDelay requires a non-null src_clk");
505  sparta_assert(dst_clk, "calculateReverseClockCrossingDelay requires a non-null dst_clk");
506  sparta_assert(src_clk->getScheduler() == dst_clk->getScheduler(),
507  "calculateReverseClockCrossingDelay requires src_clk and dst_clk to operate on "
508  "the same scheduler. src = " << src_clk->getScheduler() << " and dst = "
509  << dst_clk->getScheduler());
510  auto scheduler = src_clk->getScheduler();
511  sparta_assert(scheduler,
512  "calculateReverseClockCrossingDelay requires src_clk (" << *src_clk << ") to "
513  "have a non-null scheduler");
514 
515  sparta::Scheduler::Tick relative_ticks_before_arrival = 0;
516 
517  // 1. Snap to destination positive edge (assert match)
518  sparta::Scheduler::Tick raw_dst_clk_posedge_tick =
519  dst_arrival_tick / dst_clk->getPeriod() * dst_clk->getPeriod();
520  sparta_assert(dst_arrival_tick == raw_dst_clk_posedge_tick);
521 
522  // 2. Subtract source delay in source clock
523  // 3. Subtract destination delay in source clock
524  relative_ticks_before_arrival += src_delay + dst_delay;
525  sparta::Scheduler::Tick raw_event_sent_tick = (dst_arrival_tick - relative_ticks_before_arrival);
526 
527  // 4. Snap to previous source positive edge
528  sparta::Scheduler::Tick raw_src_clk_posedge_tick =
529  raw_event_sent_tick / src_clk->getPeriod() * src_clk->getPeriod();
530 
531  if (raw_event_sent_tick != raw_src_clk_posedge_tick) {
532  sparta_assert(raw_event_sent_tick > raw_src_clk_posedge_tick);
533  relative_ticks_before_arrival += (raw_event_sent_tick - raw_src_clk_posedge_tick);
534  }
535 
536  return relative_ticks_before_arrival;
537 }
