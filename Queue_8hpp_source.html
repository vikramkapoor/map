83  //
84  // How the Queue works internally.
85  //
86  // The queue will create storage for the next pow2 elements
87  // for efficiency. So, if the user creates a Queue of 12
88  // elements, the internal storage will be 16.
89  //
90  // There are two pointers, each represents the _physical_
91  // location in the array, not the logical.
92  //
93  // current_write_idx_ -- the insertion point used by push()
94  // current_head_idx_ -- the top of the queue, used by pop()
95  //
96  // There is one dynamically created array for the data:
97  //
98  // queue_data_ -- equal to capacity ^ next pow2
99  //
100  // At init time, current_head_idx_ == current_write_idx_. As
101  // elements are push()ed, the new objects is added and
102  // current_write_idx_ is incremented. When current_write_idx_
103  // surpasses the end of the array, it will be wrapped to the
104  // beginning -- as long as there are invalidations. The
105  // "distance" between current_write_idx_ and current_head_idx_
106  // is always <= the capacity() of the queue. The distance
107  // between the indexes represents the size().
108  //
109  // Valid iterators are given an index into the queue, which is
110  // used to retrieve the data at the given location in the
111  // queue. As the iterator is incremented, it re-validates the
112  // index given, so it will always fall into one of two
113  // locations: the next valid data element, or end()
114  //
115  // Reading/accessing the Queue from a modeler's POV is a
116  // little different. The modeler can index into the queue
117  // starting from 0 -> size(). The Queue will need to
118  // "convert" this logical index to the physical one. For
119  // example, 0 might not be the data at index 0 of the array,
120  // but instead, index 6:
121  //
122  // L P
123  // 2 0 x
124  // 3 1 x
125  // 4 2 x
126  // 5 3 <- write idx
127  // 6 4
128  // 7 5
129  // 0 6 x <- head
130  // 1 7 x
131  //
132  // Another layout of queue during runtime:
133  //
134  // L P
135  // 7 0
136  // 0 1 x <- head
137  // 1 2 x
138  // 2 3 x
139  // 3 4 x
140  // 4 5 x
141  // 5 6 x
142  // 6 7 <- write idx

732  template<typename IteratorType>
733  void decrementIterator_(IteratorType * itr) const
734  {
735  sparta_assert(itr->physical_index_ != 0, name_ <<
736  ": Iterator is not valid for decrementing");
737 
738  uint32_t physical_index = itr->physical_index_;
739 
740  // If it's the end iterator, go to the back - 1
741  if(SPARTA_EXPECT_FALSE(physical_index == invalid_index_)) {
742  const uint32_t phys_idx = decrementIndexValue_(current_write_idx_);
743  itr->physical_index_ = phys_idx;
744  itr->obj_id_ = queue_data_[phys_idx].obj_id;
745  }
746  else {
747  // See if decrementing this iterator puts into the weeds.
748  // If so, invalidate it.
749  physical_index = decrementIndexValue_(physical_index);
750  if(isValidPhysical_(physical_index)) {
751  itr->physical_index_ = physical_index;
752  itr->obj_id_ = queue_data_[physical_index].obj_id;
753  }
754  else {
