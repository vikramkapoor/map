1330  BitArray computeWriteMask_(const Definition *def) const
1331  {
1332  const auto mask_size = def->bytes;
1333  BitArray write_mask(0, mask_size);
1334  BitArray partial_mask(0, mask_size);
1335  partial_mask.fill<uint8_t>(0xff);
1336 
1337  for (auto &fdp : def->fields) {
1338  if (fdp.read_only) {
1339  const uint64_t field_size = fdp.high_bit - fdp.low_bit + 1;
1340  const uint64_t shift_down = 8 * mask_size - field_size;
1341  const uint64_t shift_up = fdp.low_bit;
1342 
1343  write_mask |= ((partial_mask >> shift_down) << shift_up);
1344  }
1345  }
1346 
1347  return ~write_mask;
1348  }
